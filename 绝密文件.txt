class exchange_code():
    def __init__(self):
        self.id = ''
        self.num = 0
        self.type = ''
        self.used = 0

    def form_code(self, code):
        data = other.encrypt_decrypt.decrypt(code, "exchange_code")
        local = data.split('&')
        self.id = local[0]
        self.num = local[1]
        self.type = local[2]
        self.used = self.Get_use()

    def to_code(self):
        code = other.encrypt_decrypt.encrypt(self.id + "&" + str(self.num) + "&" + self.type + "&" + str(self.used),
                                             "exchange_code")
        return code

    def use(self):
        cursor = conn.cursor()
        sql = "update exchange_code set used = '%s' where id = '%s'" % (1, self.id)
        conn.ping(reconnect=True)
        cursor.execute(sql)
        conn.commit()

    def Get_use(self):
        cursor = conn.cursor()
        sql = "select * from exchange_code where id = '%s'" % (self.id)
        conn.ping(reconnect=True)
        cursor.execute(sql)
        conn.commit()
        rest = cursor.fetchall()
        return rest[0][3]
